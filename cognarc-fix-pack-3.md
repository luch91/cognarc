>    cognitively-safe copy alternatives for campaign copy, landing page

>    video content with component-level findings and targeted remediation.

from pydantic import BaseModel
from typing import Optional

class CognitiveScores(BaseModel):
    cognitive_load: float
    comprehension_confidence: float
    emotional_valence: float
    trust_coherence: float
    manipulation_risk: float
    cognitive_risk: str  # LOW | MEDIUM | HIGH

class TaxonomyScores(BaseModel):
    false_urgency: float = 0
    social_proof_fabrication: float = 0
    ambiguity_exploitation: float = 0
    authority_mimicry: float = 0
    sycophantic_drift: float = 0
    obfuscation: float = 0

class RewriteRequest(BaseModel):
    original_text: str
    copy_type: str           # "campaign" | "landing_page" | "microcopy" |

    taxonomy: TaxonomyScores

class RewriteAlternative(BaseModel):

    rationale: str

class RewriteResponse(BaseModel):
    alternatives: list[RewriteAlternative]  # always 3, ranked best first
    model_used: str
    original_scores: CognitiveScores

import anthropic
import json
import os
from models import RewriteRequest, CognitiveScores, TaxonomyScores

client = anthropic.Anthropic(api_key=os.environ["ANTHROPIC_API_KEY"])

# Model routing by copy type
MODEL_MAP = {
    "campaign":      "claude-sonnet-4-6",
    "landing_page":  "claude-sonnet-4-6",
    "voiceover":     "claude-sonnet-4-6",
    "prompt":        "claude-sonnet-4-6",
    "microcopy":     "claude-haiku-4-5-20251001",
    "long_form":     "claude-opus-4-6",
}

def build_detected_patterns(taxonomy: TaxonomyScores) -> str:
    """Build a human-readable list of detected patterns above threshold."""
    LABELS = {
        "false_urgency": "artificial urgency / manufactured scarcity",
        "social_proof_fabrication": "unverified social proof / fake consensus",
        "ambiguity_exploitation": "deliberately vague language",
        "authority_mimicry": "credential inflation / authority impersonation",
        "sycophantic_drift": "excessive validation without substance",
        "obfuscation": "complexity used to hide meaning",
    }
    detected = []
    for field, label in LABELS.items():
        score = getattr(taxonomy, field, 0)
        if score > 40:
            detected.append(f"  - {label}: {score:.0f}/100")
    if not detected:
        return "  - No specific patterns above threshold"
    return "\n".join(detected)

def build_targets(scores: CognitiveScores) -> str:
    """Build target score improvements based on what's failing."""
    targets = []
    if scores.cognitive_load > 60:
        targets.append(f"  - Cognitive Load: {scores.cognitive_load:.0f} → target below 55")
    if scores.manipulation_risk > 40:
        targets.append(f"  - Manipulation Risk: {scores.manipulation_risk:.0f} → target below 35")
    if scores.comprehension_confidence < 60:
        targets.append(f"  - Comprehension Confidence: {scores.comprehension_confidence:.0f} → target above 65")
    if scores.trust_coherence < 55:
        targets.append(f"  - Trust Coherence: {scores.trust_coherence:.0f} → target above 60")
    return "\n".join(targets) if targets else "  - All scores within acceptable range"

COPY_TYPE_INSTRUCTIONS = {
    "campaign": """
- Remove or soften urgency language ("act now", "limited time", "only X left")
- Replace unsubstantiated authority claims with specific, verifiable ones or remove them
- Simplify sentence structure to reduce cognitive load
- Preserve the core value proposition exactly
- Match the original tone and brand voice
- Do not introduce new claims not present in the original""",

    "landing_page": """
- Lead with the clearest statement of value — not a question or teaser
- Remove jargon that requires domain knowledge to understand
- Each sentence should do one job: either inform, persuade, or direct
- CTA copy should describe the action, not create urgency ("See how it works" not "Act now")
- Do not change the page structure or section order — rewrite copy only""",

    "microcopy": """
- Maximum clarity in minimum words
- Button labels should describe what happens when clicked
- Error messages should say what went wrong and what to do next
- Placeholder text should give an example, not state the obvious
- Avoid negative phrasing ("don't forget" → "remember to")""",

    "voiceover": """
- Write for how people speak, not how they read
- Short sentences. One idea per sentence.
- Remove urgency language entirely — it reads as pressure in audio
- The first 8 seconds must establish value, not build to it
- Pause points matter: a comma is a breath, a full stop is a beat
- Match the pacing to the visual cuts if timing was provided""",

    "prompt": """
- Remove manipulative framing that might cause the model to produce biased outputs
- Clarify ambiguous instructions that could be interpreted multiple ways
- Reduce cognitive load by breaking compound instructions into sequential steps
- Remove sycophantic priming ("You are an expert..." type language)
- Preserve the semantic intent of the original prompt exactly""",

    "long_form": """
- Restructure if the argument is buried — lead with the conclusion
- Break sentences over 25 words into two sentences
- Replace passive voice with active voice throughout
- Remove hedging language that reduces trust coherence
- Every paragraph should earn its place — remove filler"""
}

def generate_rewrites(request: RewriteRequest) -> list[dict]:
    """Call Claude to generate 3 rewrite alternatives."""
    model = MODEL_MAP.get(request.copy_type, "claude-sonnet-4-6")
    instructions = COPY_TYPE_INSTRUCTIONS.get(request.copy_type, COPY_TYPE_INSTRUCTIONS["campaign"])
    brand_voice = f"\nBRAND VOICE NOTES:\n{request.brand_voice_notes}" if request.brand_voice_notes else ""
    length_constraint = f"\nLENGTH: Maximum {request.max_length} words per alternative." if request.max_length else ""

    prompt = f"""You are the cognitive copywriter for CognArc, an AI evaluation platform.
Your job is to rewrite copy to reduce cognitive harm while preserving intent.

ORIGINAL COPY:
\"\"\"{request.original_text}\"\"\"

TRIBE COGNITIVE SCORES:
- Cognitive Load: {request.scores.cognitive_load:.0f}/100
- Comprehension Confidence: {request.scores.comprehension_confidence:.0f}/100
- Trust Coherence: {request.scores.trust_coherence:.0f}/100
- Manipulation Risk: {request.scores.manipulation_risk:.0f}/100

DETECTED MANIPULATION PATTERNS:
{build_detected_patterns(request.taxonomy)}

IMPROVEMENT TARGETS:
{build_targets(request.scores)}

REWRITE INSTRUCTIONS for {request.copy_type.replace("_", " ")} copy:
{instructions}{brand_voice}{length_constraint}

ADDITIONAL RULES:
- Do not introduce claims not present in the original
- Do not change the fundamental message or value proposition
- Each alternative must be meaningfully different from the others
- Alternative 1 should be the most conservative edit (least changed)
- Alternative 2 should be a moderate rewrite
- Alternative 3 should be the most aggressive cognitive optimisation

Return ONLY a JSON array with exactly 3 objects. No preamble. No explanation outside JSON.
Each object must have these exact fields:
{{
  "text": "the rewritten copy",
  "rationale": "one sentence explaining the key change and why it improves the score",
  "predicted_improvement": {{
    "cognitive_load": "direction and magnitude e.g. -15 to -20 points",
    "manipulation_risk": "e.g. -40 to -50 points",
    "comprehension_confidence": "e.g. +10 to +15 points",
    "trust_coherence": "e.g. +8 to +12 points"
  }}
}}"""

    message = client.messages.create(
        model=model,
        max_tokens=1000,
        messages=[{"role": "user", "content": prompt}]
    )

    raw = message.content[0].text.strip()
    # Strip markdown fences if present
    if raw.startswith("```"):
        raw = raw.split("```")[1]
        if raw.startswith("json"):
            raw = raw[4:]
    return json.loads(raw.strip())

import httpx
import os
from models import CognitiveScores

SCORING_ENDPOINT = os.environ.get("COGNITIVE_SCORING_URL", "http://localhost:3001")

async def score_text(text: str, workspace_id: str) -> CognitiveScores:
    """Call the Cognitive Scoring Service to score a rewritten alternative."""
    async with httpx.AsyncClient(timeout=10.0) as client:
        response = await client.post(
            f"{SCORING_ENDPOINT}/score",
            json={
                "stimulus_type": "text",
                "content": text,
                "workspace_id": workspace_id,
                "options": {"manipulation_check": True}
            }
        )
        data = response.json()
        return CognitiveScores(
            cognitive_load=data["cognitive_load"],
            comprehension_confidence=data["comprehension_confidence"],
            emotional_valence=data["emotional_valence"],
            trust_coherence=data["trust_coherence"],
            manipulation_risk=data["manipulation_risk"],
            cognitive_risk=data["cognitive_risk"]
        )

import time
import asyncio
from fastapi import FastAPI, HTTPException
from models import RewriteRequest, RewriteResponse, RewriteAlternative
from rewrite_engine import generate_rewrites, MODEL_MAP
from scorer import score_text

app = FastAPI(title="CognArc Cognitive Rewrite Service")

@app.get("/health")
def health():
    return {"status": "ok", "service": "cognitive-rewrite"}

@app.post("/rewrite", response_model=RewriteResponse)
async def rewrite(request: RewriteRequest):
    start = time.time()

    # 1. Generate 3 alternatives via Claude
    try:
        raw_alternatives = generate_rewrites(request)
    except Exception as e:
        raise HTTPException(status_code=500, detail=f"Claude generation failed: {str(e)}")

    if len(raw_alternatives) != 3:
        raise HTTPException(status_code=500, detail="Expected 3 alternatives from Claude")

    # 2. Re-score each alternative via Cognitive Scoring Service
    scored = []
    for alt in raw_alternatives:
        scores = await score_text(alt["text"], request.workspace_id)
        delta = {
            "cognitive_load":           scores.cognitive_load - request.scores.cognitive_load,
            "comprehension_confidence": scores.comprehension_confidence - request.scores.comprehension_confidence,
            "trust_coherence":          scores.trust_coherence - request.scores.trust_coherence,
            "manipulation_risk":        scores.manipulation_risk - request.scores.manipulation_risk,
        }
        # Confidence: HIGH if manipulation_risk dropped > 20 and load dropped > 10
        load_drop = request.scores.cognitive_load - scores.cognitive_load
        manip_drop = request.scores.manipulation_risk - scores.manipulation_risk
        confidence = "HIGH" if (load_drop > 10 and manip_drop > 20) \
                    else "MEDIUM" if (load_drop > 5 or manip_drop > 10) \
                    else "LOW"
        scored.append(RewriteAlternative(
            text=alt["text"],
            rationale=alt["rationale"],
            scores=scores,
            score_delta=delta,
            confidence=confidence
        ))

    # 3. Rank by overall improvement (lower manipulation + lower load + higher comprehension)
    def improvement_score(alt: RewriteAlternative) -> float:
        d = alt.score_delta
        return (
            -d["manipulation_risk"] * 2.0 +   # most important
            -d["cognitive_load"] * 1.5 +
             d["comprehension_confidence"] * 1.0 +
             d["trust_coherence"] * 0.8
        )

    scored.sort(key=improvement_score, reverse=True)

    return RewriteResponse(
        alternatives=scored,
        model_used=MODEL_MAP.get(request.copy_type, "claude-sonnet-4-6"),
        original_scores=request.scores,
        processing_time_ms=int((time.time() - start) * 1000)
    )

export interface RewriteRequest {
  originalText: string
  copyType: 'campaign' | 'landing_page' | 'microcopy' | 'voiceover' | 'prompt' | 'long_form'
  scores: {
    cognitiveLoad: number
    comprehensionConfidence: number

    socialProofFabrication?: number

    authorityMimicry?: number
    sycophantidDrift?: number

  workspaceId: string

export interface RewriteAlternative {
  text: string
  rationale: string
  scores: RewriteRequest['scores']
  scoreDelta: {
    cognitiveLoad: number
    comprehensionConfidence: number
    trustCoherence: number
    manipulationRisk: number
  }

export interface RewriteResponse {
  alternatives: RewriteAlternative[]  // 3 items, ranked best first
  modelUsed: string
  originalScores: RewriteRequest['scores']
  processingTimeMs: number

const REWRITE_URL = import.meta.env.VITE_COGNITIVE_REWRITE_URL ?? 'http://localhost:3006'

export async function requestRewrites(request: RewriteRequest): Promise<RewriteResponse> {
  // Convert camelCase to snake_case for the Python backend
  const payload = {
    original_text: request.originalText,
    copy_type: request.copyType,
    scores: {
      cognitive_load: request.scores.cognitiveLoad,
      comprehension_confidence: request.scores.comprehensionConfidence,
      emotional_valence: request.scores.emotionalValence,
      trust_coherence: request.scores.trustCoherence,
      manipulation_risk: request.scores.manipulationRisk,
      cognitive_risk: request.scores.cognitiveRisk,
    },
    taxonomy: {
      false_urgency: request.taxonomy?.falseUrgency ?? 0,
      social_proof_fabrication: request.taxonomy?.socialProofFabrication ?? 0,
      ambiguity_exploitation: request.taxonomy?.ambiguityExploitation ?? 0,
      authority_mimicry: request.taxonomy?.authorityMimicry ?? 0,
      sycophantic_drift: request.taxonomy?.sycophantidDrift ?? 0,
      obfuscation: request.taxonomy?.obfuscation ?? 0,
    },
    brand_voice_notes: request.brandVoiceNotes,
    max_length: request.maxLength,
    workspace_id: request.workspaceId,
  }

  const res = await fetch(`${REWRITE_URL}/rewrite`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(payload),
  })

  if (!res.ok) {
    throw new Error(`Rewrite service error: ${res.status}`)
  }

  const data = await res.json()

  // Convert snake_case response back to camelCase
  return {
    alternatives: data.alternatives.map((alt: any) => ({
      text: alt.text,
      rationale: alt.rationale,
      confidence: alt.confidence,
      scores: {
        cognitiveLoad: alt.scores.cognitive_load,
        comprehensionConfidence: alt.scores.comprehension_confidence,
        emotionalValence: alt.scores.emotional_valence,
        trustCoherence: alt.scores.trust_coherence,
        manipulationRisk: alt.scores.manipulation_risk,
        cognitiveRisk: alt.scores.cognitive_risk,
      },
      scoreDelta: {
        cognitiveLoad: alt.score_delta.cognitive_load,
        comprehensionConfidence: alt.score_delta.comprehension_confidence,
        trustCoherence: alt.score_delta.trust_coherence,
        manipulationRisk: alt.score_delta.manipulation_risk,
      }
    })),
    modelUsed: data.model_used,
    originalScores: request.scores,
    processingTimeMs: data.processing_time_ms,
  }
}

from fastapi import FastAPI
from pydantic import BaseModel
from typing import Optional
import hashlib, time

app = FastAPI(title="CognArc Video Cognitive Analysis")

class VideoAnalysisRequest(BaseModel):
    filename: str
    file_size_bytes: int
    duration_estimate_seconds: Optional[float] = 30.0
    workspace_id: str

class MomentFinding(BaseModel):
    timestamp_start: float
    timestamp_end: float
    component: str      # "opening_hook" | "voiceover" | "scene_transition" |
                        # "product_demo" | "cta" | "outro"
    cognitive_load: float
    manipulation_risk: float
    trust_coherence: float
    attention_engagement: float
    severity: str       # "critical" | "warning" | "ok"
    finding: str        # plain-language description of what's wrong
    recommendation: str # what to do about it
    voiceover_segment: Optional[str] = None  # text to rewrite if voiceover issue

class VideoAnalysisResponse(BaseModel):
    filename: str
    duration_seconds: float
    overall_scores: dict
    findings: list[MomentFinding]
    component_summary: dict
    rewrite_candidates: list[str]  # voiceover segments eligible for rewrite
    analysis_mode: str  # "mock" | "tribe-local" | "tribe-gcp"

@app.get("/health")
def health():
    return {"status": "ok", "service": "video-analysis", "mode": "mock"}

@app.post("/analyze", response_model=VideoAnalysisResponse)
async def analyze(request: VideoAnalysisRequest):
    # Deterministic mock: derive scores from filename hash so the same
    # file always produces the same analysis
    seed = int(hashlib.md5(request.filename.encode()).hexdigest()[:8], 16)
    duration = request.duration_estimate_seconds or 30.0

    # Generate 4-6 moment findings covering the video structure
    findings = [
        MomentFinding(
            timestamp_start=0.0,
            timestamp_end=min(8.0, duration * 0.25),
            component="opening_hook",
            cognitive_load=78 + (seed % 10),
            manipulation_risk=22,
            trust_coherence=61,
            attention_engagement=72,
            severity="warning",
            finding="Visual and text elements competing for attention simultaneously in the first 8 seconds. Cognitive load spike before value proposition is established.",
            recommendation="Lead with visual only for the first 3 seconds before introducing text overlays. Establish the value proposition before adding supporting elements.",
        ),
        MomentFinding(
            timestamp_start=duration * 0.3,
            timestamp_end=duration * 0.35,
            component="voiceover",
            cognitive_load=45,
            manipulation_risk=84 + (seed % 8),
            trust_coherence=38,
            attention_engagement=68,
            severity="critical",
            finding="False urgency language detected in voiceover. Manufactured scarcity phrases trigger limbic activation without prefrontal engagement — bypassing rational evaluation.",
            recommendation="Replace urgency language with benefit-focused framing. See rewrite suggestions below.",
            voiceover_segment="Act now — experts unanimously agree this is a limited time opportunity. Don't miss out.",
        ),
        MomentFinding(
            timestamp_start=duration * 0.5,
            timestamp_end=duration * 0.55,
            component="scene_transition",
            cognitive_load=52,
            manipulation_risk=28,
            trust_coherence=39,
            attention_engagement=55,
            severity="warning",
            finding=f"Trust coherence drops {18 + (seed % 8)} points at this transition. Audio message is benefit-focused while the visual introduces urgency/pressure imagery. Misalignment breaks trust.",
            recommendation="Align the visual to the benefit message being spoken, or cut this transition and connect the benefit moment directly to the product demo.",
        ),
        MomentFinding(
            timestamp_start=duration * 0.6,
            timestamp_end=duration * 0.8,
            component="product_demo",
            cognitive_load=71,
            manipulation_risk=18,
            trust_coherence=64,
            attention_engagement=44,
            severity="warning",
            finding="Low attention engagement during product demo. No clear focal point — viewer attention is dispersed across the screen. Visual complexity score is high.",
            recommendation="Add motion or highlight to direct viewer attention to the key UI element. Reduce on-screen elements by ~40% per frame. One focus point per scene.",
        ),
        MomentFinding(
            timestamp_start=duration * 0.85,
            timestamp_end=duration,
            component="cta",
            cognitive_load=48,
            manipulation_risk=55 + (seed % 15),
            trust_coherence=52,
            attention_engagement=78,
            severity="warning",
            finding="CTA uses urgency framing that partially undermines the trust built during the product demo. Moderate manipulation risk from pressure language.",
            recommendation="Replace the CTA urgency copy with a low-pressure action phrase. The product demo did the work — the CTA should confirm, not pressure.",
            voiceover_segment="Sign up today before this offer expires.",
        ),
    ]

    # Overall scores (weighted average)
    overall_manipulation = sum(f.manipulation_risk for f in findings) / len(findings)
    overall_load = sum(f.cognitive_load for f in findings) / len(findings)
    overall_trust = sum(f.trust_coherence for f in findings) / len(findings)
    overall_attention = sum(f.attention_engagement for f in findings) / len(findings)

    cognitive_risk = "HIGH" if overall_manipulation > 60 or overall_load > 70 \
        else "MEDIUM" if overall_manipulation > 40 or overall_load > 55 \
        else "LOW"

        "product_demo": {"severity": "warning", "primary_issue": "attention_engagement"},
        "cta": {"severity": "warning", "primary_issue": "manipulation_risk"},
    }

    return VideoAnalysisResponse(
        filename=request.filename,
        duration_seconds=duration,
        overall_scores={
            "cognitive_load": round(overall_load, 1),
            "manipulation_risk": round(overall_manipulation, 1),
            "trust_coherence": round(overall_trust, 1),
            "attention_engagement": round(overall_attention, 1),
            "cognitive_risk": cognitive_risk,
        },
        findings=findings,
        component_summary=component_summary,
        rewrite_candidates=rewrite_candidates,
        analysis_mode="mock",
    )

## Architecture Reminder

```
User submits copy / video
        ↓
Cognitive Scoring Service (TRIBE/mock)
        ↓ scores breach threshold
Cognitive Rewrite Service
        ↓ calls Claude Sonnet/Haiku/Opus
Claude generates 3 alternatives
        ↓
Cognitive Scoring Service re-scores each alternative
        ↓
Alternatives ranked by improvement score
        ↓
Human reviews → selects or rejects
        ↓ (if selected)
AppContext updated → Audit log entry written

TRIBE evaluates. Claude generates. Humans decide.
```
